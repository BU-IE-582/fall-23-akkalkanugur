# function for run all 4 approaches, find best parameters and apply cross-validation on test and training set
def run_algorithms(X_train, X_test, y_train, y_test):
    # Create a RandomForestClassifier
    classifier = RandomForestClassifier(n_estimators=500, min_samples_leaf=5,random_state=42)
    classifier.fit(X_train, y_train)
    y_pred = classifier.predict(X_test)

    # Nearest Neighbor (NN) Approaches
    nn_classifier = KNeighborsClassifier(n_neighbors=5)
    nn_classifier.fit(X_train, y_train)
    nn_predictions = nn_classifier.predict(X_test)

    # Decision Trees (DT) Classifier
    dt_classifier = DecisionTreeClassifier(random_state=42)
    dt_classifier.fit(X_train, y_train)
    dt_class_predictions = dt_classifier.predict(X_test)

    # Decision Trees (DT) Regressor
    dt_regressor = DecisionTreeRegressor()
    dt_regressor.fit(X_train, y_train)
    dt_regress_predictions = dt_regressor.predict(X_test)

    # Gradient Boosted Trees (GBT)
    gbt_classifier = GradientBoostingClassifier(n_estimators=100, random_state=42)
    gbt_classifier.fit(X_train, y_train)
    gbt_predictions = gbt_classifier.predict(X_test)

##--##    
    # Search best parameters for RandomForestClassifier
    print("* 1.a) Search best parameters for RandomForestClassifier")
    param_grid = {
    'max_features': [1,5,20,30,50,100,200, 'auto', 'sqrt', 'log2'],       # Number of features to consider at each split
    'criterion': ['gini', 'entropy']  # Function to measure the quality of a split
    }

    # Create a GridSearchCV object and Fit the model to the data
    grid_search = GridSearchCV(classifier, param_grid,scoring='accuracy') #scoring='roc_auc'
    grid_search.fit(X_train, y_train)

    # Get the best model from the grid search and Make predictions on the test set
    best_rf_model = grid_search.best_estimator_
    y_pred = best_rf_model.predict(X_train)
    # Access the best hyperparameters
    best_params = grid_search.best_params_
    print('Best Hyperparameters:', best_params)
    print('Best Score:',grid_search.scoring,":","{:.2%}".format(grid_search.best_score_))
    
    print("* 1.b) Cross-validation scores for the training set for RandomForestClassifier")
    cv_scores_train = cross_val_score(classifier, X_train, y_train, cv=5, scoring='accuracy')

    # Predict on the test set & Calculate accuracy on the test set
    y_pred_test = classifier.predict(X_test)
    accuracy_test = accuracy_score(y_test, y_pred_test)

    plt.figure(figsize=(5, 3))
    plt.title("Cross Validation Plot for RandomForestClassifier")
    x_values = np.arange(1, len(cv_scores_train)+1) 
    plt.plot(x_values,cv_scores_train,marker='o', label="cv on train-set")
    plt.plot(x_values,[accuracy_test]*5,marker='x', label="accuracy on test-set")
    plt.plot(x_values,[grid_search.best_score_]*5,marker='x', label="accuracy on train-set")
    plt.legend()
    plt.show

    print("Cross-Validation Scores on Training Set:", cv_scores_train)
    print("Accuracy on Training Set:", grid_search.best_score_)
    print("Accuracy on Test Set:", accuracy_test)

##--##

    # Search best parameters for Nearest Neighbor (NN)
    print("\n* 2.a) Search best parameters for Nearest Neighbor (NN)")
    param_grid = {
        'n_neighbors': [1, 5, 10, 25, 50],  # Adjust the number of neighbors as needed
        'metric': ['euclidean', 'manhattan'],  # Distance measures to consider
    }

    # Create a GridSearchCV object and Fit the model to the data
    grid_search = GridSearchCV(nn_classifier, param_grid,scoring='accuracy') #scoring='roc_auc'
    grid_search.fit(X_train, y_train)

    # Get the best model from the grid search and Make predictions on the test set
    best_model = grid_search.best_estimator_
    y_pred = best_model.predict(X_train)
    # Access the best hyperparameters

    print('Best Hyperparameters:', best_params)
    print('Best Score:',grid_search.scoring,":","{:.2%}".format(grid_search.best_score_))
    
    # Perform cross-validation and get the cross-validation scores for the training set
    print("* 2.b) Cross-validation scores for the training set for Nearest Neighbor (NN)")
    cv_scores_train = cross_val_score(nn_classifier, X_train, y_train, cv=5, scoring='accuracy')

    # Predict on the test set & Calculate accuracy on the test set
    y_pred_test = nn_classifier.predict(X_test)
    accuracy_test = accuracy_score(y_test, y_pred_test)

    plt.figure(figsize=(5, 3))
    plt.title("Cross Validation Plot for Nearest Neighbor (NN)")
    x_values = np.arange(1, len(cv_scores_train)+1) 
    plt.plot(x_values,cv_scores_train,marker='o', label="cv on train-set")
    plt.plot(x_values,[accuracy_test]*5,marker='x', label="accuracy on test-set")
    plt.plot(x_values,[grid_search.best_score_]*5,marker='x', label="accuracy on train-set")

##--##
    
    # Search best parameters for Decision Trees (DT) Classifier
    print("\n* 3.a) Search best parameters for Decision Trees (DT) Classifier")
    param_grid = {
        'min_samples_leaf': [5, 10, 20, 50, 100],  # Adjust the values as needed
    }

    # Create a GridSearchCV object and Fit the model to the data
    grid_search = GridSearchCV(dt_classifier, param_grid,scoring='neg_mean_squared_error') #scoring='roc_auc'
    grid_search.fit(X_train, y_train)

    # Get the best model from the grid search and Make predictions on the test set
    best_model = grid_search.best_estimator_
    y_pred = best_model.predict(X_train)
    # Access the best hyperparameters
